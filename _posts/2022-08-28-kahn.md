const int MM = 10;   
vi adj[MM];   
int in[MM];   
   
void addEdge(int u, int v){   
    adj[u].pb(v);   
    in[v]++;   
}   
   
int main(){   
    ios::sync_with_stdio(0);   
    cin.tie(0);   
    addEdge(1, 3);   
    addEdge(1, 5);   
    addEdge(1, 6);   
    addEdge(2, 5);   
    addEdge(2, 6);   
    addEdge(3, 4);   
    addEdge(3, 5);   
    addEdge(5, 6);   
   
    priority_queue<int, vi, greater<int>> q;   
    for(int i = 1; i <= 6; i++){   
        if(in[i] == 0) q.push(i);   
    }   
   
    while(!q.empty()){   
        int c = q.top(); q.pop();   
        cout << c << ' ';   
        for(int x : adj[c]){   
            if(--in[x] == 0) q.push(x);   
        }   
    }   
       
    return 0;   
}
